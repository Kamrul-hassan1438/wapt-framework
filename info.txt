Via the API:
bash# Start the server
python cli.py server

# Generate report via API
curl -X POST http://localhost:8000/api/reports/generate \
  -H "Content-Type: application/json" \
  -d '{"scan_id": "<scan_id>", "formats": ["html","pdf","json","markdown"]}'

# List all generated reports
curl http://localhost:8000/api/reports/list

# Quick summary (no file generated)
curl http://localhost:8000/api/reports/<scan_id>/summary

# Download a specific report
curl -O http://localhost:8000/api/reports/download/WAPT-8A6D0172-20260409-155533.pdf
